Write a C program depth first search (DFS) using array.

#include<stdio.h>

int g[10][10],visited[10],n;

void DFS(int i) {

    int j;

    printf("\n%d,i");

    visited[i]=1;

    for (j=0;j<n;j++){

        if (!visited[j]&g[i][j]==1)

        DFS(j);

    }

}

void main() {

    int i,j;

    printf("Enter number of vertices:");

    scanf("%d",&n);

    printf("\nEnter adjancency matrix of the graph");

    for (i=0;i<n;i++){

        for (j=0;j<n;j++){

            scanf("%d",&g[i][j]);

        }

    }

    for(i=0;i<n;i++){

        visited[i]=0;

          DFS(0);

    }

}
